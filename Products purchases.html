query = """
SELECT
    TO_CHAR(purchase_date, 'TMMonth') AS ,
    EXTRACT(DAY FROM purchase_date) AS ,
    COUNT(*) AS 
FROM products
WHERE
    tariff = 'hard'
    AND EXTRACT(YEAR FROM purchase_date) = 2024
    AND gender = 'm'
GROUP BY 1, 2, EXTRACT(MONTH FROM purchase_date)
ORDER BY EXTRACT(MONTH FROM purchase_date), ;
"""

df = pd.read_sql(query, engine)

grouped = df.groupby('weekday')['']

def find_outliers(group):
    q1 = group.quantile(0.25)
    q3 = group.quantile(0.75)
    iqr = q3 - q1
    lower = q1 - 1.5 * iqr
    upper = q3 + 1.5 * iqr
    return group[(group < lower) | (group > upper)]
